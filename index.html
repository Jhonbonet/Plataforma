// API de autenticación - Versión corregida para la estructura de la base de datos asenorte.sql export default async function handler(req, res) { // Configuración de CORS const allowedOrigins = ['http://localhost', 'http://localhost:3000', 'https://tu-dominio.com']; const origin = req.headers.origin; if (allowedOrigins.includes(origin)) { res.setHeader('Access-Control-Allow-Origin', origin); } res.setHeader('Access-Control-Allow-Methods', 'GET, POST, OPTIONS'); res.setHeader('Access-Control-Allow-Headers', 'Content-Type, Authorization'); res.setHeader('Access-Control-Allow-Credentials', 'true'); // Manejar preflight if (req.method === 'OPTIONS') { return res.status(200).end(); } // Solo permitir POST para login if (req.method !== 'POST') { return res.status(405).json({ error: 'Método no permitido' }); } const { documento, tipoDocumento, programa } = req.body; // Validaciones básicas if (!documento) { return res.status(400).json({ error: 'El número de documento es requerido' }); } if (!programa) { return res.status(400).json({ error: 'Debe seleccionar un programa' }); } try { // Configuración de la conexión a MySQL const mysql = require('mysql2/promise'); // Crear conexión a la base de datos - Ajusta estos valores a tu configuración local const connection = await mysql.createConnection({ host: 'localhost', port: 3306, user: 'root', // Cambia por tu usuario de MySQL password: '', // Cambia por tu contraseña de MySQL database: 'asenorte' // Nombre de tu base de datos }); // PRIMERA ESTRATEGIA: Buscar en la tabla fac_facturas (la más completa) // ya que contiene CLIENTE_FACTURA que es el ID del estudiante let query = ` SELECT ff.ID_FACTURA, ff.CLIENTE_FACTURA as documento, ff.TIPO_FACTURA, ff.CONSECUTIVO_FACTURA, ff.FECHA_FACTURA, ff.SUBTOTAL_FACTURA, ff.TOTAL_FACTURA, ff.DESCRIPCION_FACTURA, ff.VENDEDOR_FACTURA, ff.TERMINO_FACTURA, ff.VENCIMIENTO_FACTURA, ff.TIPOPAGO_FACTURA, 'FACTURA' as tipo_registro FROM fac_facturas ff WHERE ff.CLIENTE_FACTURA = ? UNION ALL SELECT fc.ID_CREDITO as ID_FACTURA, fc.CLIENTE_CREDITO as documento, 'CREDITO' as TIPO_FACTURA, fc.CONSECUTIVO_CREDITO as CONSECUTIVO_FACTURA, fc.FECHA_CREDITO as FECHA_FACTURA, fc.VALOR_CREDITO as SUBTOTAL_FACTURA, fc.VALOR_CREDITO as TOTAL_FACTURA, fc.CONCEPTO_CREDITO as DESCRIPCION_FACTURA, fc.VENDEDOR_CREDITO as VENDEDOR_FACTURA, NULL as TERMINO_FACTURA, NULL as VENCIMIENTO_FACTURA, NULL as TIPOPAGO_FACTURA, 'CREDITO' as tipo_registro FROM fac_creditos fc WHERE fc.CLIENTE_CREDITO = ? UNION ALL SELECT fr.ID_RECIBO as ID_FACTURA, fr.CLIENTE_RECIBO as documento, 'RECIBO' as TIPO_FACTURA, fr.CONSECUTIVO_RECIBO as CONSECUTIVO_FACTURA, fr.FECHA_RECIBO as FECHA_FACTURA, fr.VALOR_RECIBO as SUBTOTAL_FACTURA, fr.VALOR_RECIBO as TOTAL_FACTURA, fr.CONCEPTO_RECIBO as DESCRIPCION_FACTURA, fr.VENDEDOR_RECIBO as VENDEDOR_FACTURA, NULL as TERMINO_FACTURA, NULL as VENCIMIENTO_FACTURA, NULL as TIPOPAGO_FACTURA, 'RECIBO' as tipo_registro FROM fac_recibos fr WHERE fr.CLIENTE_RECIBO = ? ORDER BY FECHA_FACTURA DESC LIMIT 50 `; const [rows] = await connection.execute(query, [documento, documento, documento]); // Cerrar conexión await connection.end(); // Si encontró registros, el estudiante existe if (rows && rows.length > 0) { // Obtener información adicional del primer registro encontrado const primerRegistro = rows[0]; // Construir objeto de respuesta con información del estudiante const estudianteInfo = { documento: documento, tipoDocumento: tipoDocumento || 'Cédula', programa: programa, nombre: `Estudiante ${documento}`, // Nombre por defecto si no está en BD tieneRegistros: true, totalRegistros: rows.length, ciudad: 'Santa Marta', institucion: 'ASENORTE - Asesorías del Norte', ultimosRegistros: rows.slice(0, 10).map(row => ({ tipo: row.tipo_registro, consecutivo: row.CONSECUTIVO_FACTURA, fecha: row.FECHA_FACTURA, valor: row.SUBTOTAL_FACTURA || row.TOTAL_FACTURA, descripcion: row.DESCRIPCION_FACTURA, vendedor: row.VENDEDOR_FACTURA })) }; // Intentar obtener el nombre del estudiante desde fac_recibos // (podría estar en VENDEDOR_RECIBO cuando es un registro de pago) const registroConNombre = rows.find(r => r.VENDEDOR_FACTURA && r.VENDEDOR_FACTURA !== 'Administrador' && r.VENDEDOR_FACTURA !== 'NELLY MARIA ÁLVAREZ BORREGO' && r.VENDEDOR_FACTURA !== 'Lucila Isabel Fernandez de Castro Dangond'); if (registroConNombre && registroConNombre.VENDEDOR_FACTURA) { estudianteInfo.nombre = registroConNombre.VENDEDOR_FACTURA; } return res.status(200).json({ success: true, message: 'Autenticación exitosa', estudiante: estudianteInfo, registros: rows }); } else { // SEGUNDA ESTRATEGIA: Buscar en recibos específicos para confirmar existencia const connection2 = await mysql.createConnection({ host: 'localhost', port: 3306, user: 'root', password: '', database: 'asenorte' }); // Buscar en recibos const [recibos] = await connection2.execute( `SELECT COUNT(*) as total FROM fac_recibos WHERE CLIENTE_RECIBO = ?`, [documento] ); // Buscar en créditos const [creditos] = await connection2.execute( `SELECT COUNT(*) as total FROM fac_creditos WHERE CLIENTE_CREDITO = ?`, [documento] ); // Buscar en facturas const [facturas] = await connection2.execute( `SELECT COUNT(*) as total FROM fac_facturas WHERE CLIENTE_FACTURA = ?`, [documento] ); await connection2.end(); const existe = (recibos[0]?.total || 0) > 0 || (creditos[0]?.total || 0) > 0 || (facturas[0]?.total || 0) > 0; if (existe) { // El estudiante existe pero no tiene facturas visibles en la primera consulta return res.status(200).json({ success: true, message: 'Autenticación exitosa - Estudiante registrado', estudiante: { documento: documento, tipoDocumento: tipoDocumento || 'Cédula', programa: programa, tieneRegistros: true, totalRecibos: recibos[0]?.total || 0, totalCreditos: creditos[0]?.total || 0, totalFacturas: facturas[0]?.total || 0 } }); } else { return res.status(401).json({ error: 'Documento no encontrado', message: 'No se encontraron registros para este número de documento en el sistema.' }); } } } catch (error) { console.error('Error en autenticación:', error); // En caso de error, devolver respuesta de diagnóstico return res.status(500).json({ error: 'Error interno del servidor', message: error.message, stack: process.env.NODE_ENV === 'development' ? error.stack : undefined }); } }
